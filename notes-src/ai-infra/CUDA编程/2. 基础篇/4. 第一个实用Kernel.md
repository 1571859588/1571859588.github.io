## 一句话结论

CUDA 编程五步流程：`cudaMalloc` → `cudaMemcpy(H2D)` → `kernel<<<grid,block>>>()` → `cudaMemcpy(D2H)` → `cudaFree`。向量加法是最简单的并行问题，但它建立起 CUDA 编程的完整框架——后续所有高级 Kernel 都是在这个框架上增加复杂度。

## 核心概念（新人友好讲解）

### 1. 从 CPU 到 GPU：完整开发流程

将计算任务从 CPU 搬到 GPU 就像把工作从一个能干的员工交给一支千人团队——你需要准备材料、分配任务、等待完成、收回成果。对应到 CUDA 编程：

```
1. 分配 GPU 内存     cudaMalloc()
       │
       ▼
2. CPU→GPU 传数据   cudaMemcpy(..., HostToDevice)
       │
       ▼
3. 启动 Kernel       kernel<<<grid, block>>>(args)
       │
       ▼
4. GPU→CPU 取结果   cudaMemcpy(..., DeviceToHost)
       │
       ▼
5. 释放 GPU 内存     cudaFree()
```

| 步骤 | API | 说明 |
|------|-----|------|
| 分配 GPU 内存 | `cudaMalloc()` | 类似 CPU 的 `malloc()` |
| Host→Device 传输 | `cudaMemcpy(..., HostToDevice)` | CPU 数据拷贝到 GPU |
| 启动 Kernel | `kernel<<<grid, block>>>(args)` | GPU 并行执行 |
| Device→Host 传输 | `cudaMemcpy(..., DeviceToHost)` | 结果拷回 CPU |
| 释放 GPU 内存 | `cudaFree()` | 类似 CPU 的 `free()` |

### 2. 向量加法 Kernel

向量加法是最简单的并行问题——每个元素的计算完全独立，天然适合 GPU。给定两个长度为 N 的向量 A 和 B，计算：

```
C[i] = A[i] + B[i],  i = 0, 1, ..., N-1
```

#### 2.1 Kernel 函数

```cpp
__global__ void vectorAdd(const float* A, const float* B, float* C, int N) {
    // 计算全局线程索引
    int idx = blockIdx.x * blockDim.x + threadIdx.x;

    // 边界检查：防止越界访问
    if (idx < N) {
        C[idx] = A[idx] + B[idx];
    }
}
```

**代码解析**（新人必看）：
- `const float* A, B`：只读输入，用 `const` 让编译器和硬件知道可以走只读缓存
- `int idx = blockIdx.x * blockDim.x + threadIdx.x`：全局索引公式（1.2 篇讲过）
- `if (idx < N)`：边界检查，必须加
- `C[idx] = A[idx] + B[idx]`：每个线程独立完成一个元素

#### 2.2 为什么需要边界检查（新人易忽略）

假设 N = 1000，Block 大小 = 256：
- 需要 Block 数 = ⌈1000/256⌉ = 4
- 总线程数 = 4 × 256 = 1024
- **多出 24 个线程**的索引为 1000~1023，超出数组范围

没有 `if (idx < N)` 检查，这些线程会访问非法内存地址，导致未定义行为或程序崩溃。

### 完整可运行代码

将内存管理、Kernel、错误检查、性能测量整合为一个完整程序：

```cpp
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <chrono>
#include <cuda_runtime.h>

// 错误检查宏
#define CUDA_CHECK(call)                                                      \
    do {                                                                       \
        cudaError_t err = call;                                                \
        if (err != cudaSuccess) {                                              \
            fprintf(stderr, "CUDA error at %s:%d - %s\n",                     \
                    __FILE__, __LINE__, cudaGetErrorString(err));              \
            exit(EXIT_FAILURE);                                                \
        }                                                                      \
    } while (0)

// GPU Kernel
__global__ void vectorAdd(const float* A, const float* B, float* C, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        C[idx] = A[idx] + B[idx];
    }
}

// CPU 基线
void vectorAddCPU(const float* A, const float* B, float* C, int N) {
    for (int i = 0; i < N; i++) {
        C[i] = A[i] + B[i];
    }
}

int main() {
    const int N = 1 << 24;  // 约 1600 万元素
    const size_t size = N * sizeof(float);
    printf("Vector size: %d elements (%.1f MB)\n", N, size / (1024.0 * 1024.0));

    // ========== 分配主机内存 ==========
    float* h_A = (float*)malloc(size);
    float* h_B = (float*)malloc(size);
    float* h_C_cpu = (float*)malloc(size);
    float* h_C_gpu = (float*)malloc(size);

    // 初始化数据
    for (int i = 0; i < N; i++) {
        h_A[i] = sinf(i * 0.001f);
        h_B[i] = cosf(i * 0.001f);
    }

    // ========== CPU 计算 ==========
    auto cpu_start = std::chrono::high_resolution_clock::now();
    vectorAddCPU(h_A, h_B, h_C_cpu, N);
    auto cpu_end = std::chrono::high_resolution_clock::now();
    float cpu_ms = std::chrono::duration<float, std::milli>(cpu_end - cpu_start).count();
    printf("CPU time: %.4f ms\n", cpu_ms);

    // ========== GPU 计算 ==========
    float *d_A, *d_B, *d_C;
    CUDA_CHECK(cudaMalloc(&d_A, size));
    CUDA_CHECK(cudaMalloc(&d_B, size));
    CUDA_CHECK(cudaMalloc(&d_C, size));

    // 创建 CUDA Event 计时器
    cudaEvent_t start, stop, kernel_start, kernel_stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);
    cudaEventCreate(&kernel_start);
    cudaEventCreate(&kernel_stop);

    // 端到端计时（含传输）
    cudaEventRecord(start);

    CUDA_CHECK(cudaMemcpy(d_A, h_A, size, cudaMemcpyHostToDevice));
    CUDA_CHECK(cudaMemcpy(d_B, h_B, size, cudaMemcpyHostToDevice));

    int blockSize = 256;
    int gridSize = (N + blockSize - 1) / blockSize;

    // 仅 Kernel 计时
    cudaEventRecord(kernel_start);
    vectorAdd<<<gridSize, blockSize>>>(d_A, d_B, d_C, N);
    cudaEventRecord(kernel_stop);

    CUDA_CHECK(cudaGetLastError());
    CUDA_CHECK(cudaMemcpy(h_C_gpu, d_C, size, cudaMemcpyDeviceToHost));

    cudaEventRecord(stop);
    cudaEventSynchronize(stop);

    float gpu_total_ms = 0, kernel_ms = 0;
    cudaEventElapsedTime(&gpu_total_ms, start, stop);
    cudaEventElapsedTime(&kernel_ms, kernel_start, kernel_stop);

    printf("GPU kernel time: %.4f ms\n", kernel_ms);
    printf("GPU total time (with memcpy): %.4f ms\n", gpu_total_ms);
    printf("Speedup (kernel only): %.1fx\n", cpu_ms / kernel_ms);
    printf("Speedup (end-to-end): %.1fx\n", cpu_ms / gpu_total_ms);

    // ========== 结果验证 ==========
    int errors = 0;
    for (int i = 0; i < N; i++) {
        if (fabsf(h_C_gpu[i] - h_C_cpu[i]) > 1e-5f) {
            if (errors < 5) {
                printf("Mismatch at %d: CPU=%.6f, GPU=%.6f\n",
                       i, h_C_cpu[i], h_C_gpu[i]);
            }
            errors++;
        }
    }
    if (errors == 0) {
        printf("Verification PASSED!\n");
    } else {
        printf("Verification FAILED: %d mismatches\n", errors);
    }

    // ========== 计算有效带宽 ==========
    // 向量加法读 2 个 float、写 1 个 float = 12 bytes/element
    float bandwidth_gb = (3.0f * size) / (kernel_ms * 1e-3f) / 1e9f;
    printf("Effective bandwidth: %.1f GB/s\n", bandwidth_gb);

    // ========== 清理 ==========
    cudaEventDestroy(start);
    cudaEventDestroy(stop);
    cudaEventDestroy(kernel_start);
    cudaEventDestroy(kernel_stop);
    CUDA_CHECK(cudaFree(d_A));
    CUDA_CHECK(cudaFree(d_B));
    CUDA_CHECK(cudaFree(d_C));
    free(h_A);
    free(h_B);
    free(h_C_cpu);
    free(h_C_gpu);

    return 0;
}
```

### CUDA 错误检查详解（新人必看）

CUDA API 调用可能失败（内存不足、参数错误等），生产代码必须检查返回值。

#### 错误检查宏

```cpp
#define CUDA_CHECK(call)                                                      \
    do {                                                                       \
        cudaError_t err = call;                                                \
        if (err != cudaSuccess) {                                              \
            fprintf(stderr, "CUDA error at %s:%d - %s\n",                     \
                    __FILE__, __LINE__, cudaGetErrorString(err));              \
            exit(EXIT_FAILURE);                                                \
        }                                                                      \
    } while (0)
```

#### 使用方式

```cpp
CUDA_CHECK(cudaMalloc(&d_A, size));
CUDA_CHECK(cudaMemcpy(d_A, h_A, size, cudaMemcpyHostToDevice));

// Kernel 启动的错误检查需要两步
vectorAdd<<<gridSize, blockSize>>>(d_A, d_B, d_C, N);
CUDA_CHECK(cudaGetLastError());          // 检查启动参数是否正确
CUDA_CHECK(cudaDeviceSynchronize());     // 检查执行过程是否有错误
```

> 🔧 **为什么 kernel 检查要两步**：kernel 启动是异步的，`cudaGetLastError()` 只捕获**启动时**错误（如配置非法），`cudaDeviceSynchronize()` 后才能捕获**执行时**错误（如越界访问）。

#### 常见错误码（新人速查）

| 错误码 | 含义 | 常见原因 |
|--------|------|---------|
| `cudaErrorMemoryAllocation` | 内存分配失败 | 显存不足 |
| `cudaErrorInvalidValue` | 参数无效 | size=0 或空指针 |
| `cudaErrorInvalidConfiguration` | 启动配置无效 | Block 超过 1024 线程 |
| `cudaErrorIllegalAddress` | 非法内存访问 | 越界/野指针 |
| `cudaErrorLaunchTimeout` | Kernel 执行超时 | 死循环或计算量过大 |

### 性能测量：CUDA Event 计时（新人重点）

CUDA Event 是 GPU 端精确计时的标准方法，精度达微秒级：

```cpp
cudaEvent_t start, stop;
cudaEventCreate(&start);
cudaEventCreate(&stop);

// 计时开始
cudaEventRecord(start);

// 要计时的操作
vectorAdd<<<gridSize, blockSize>>>(d_A, d_B, d_C, N);

// 计时结束
cudaEventRecord(stop);
cudaEventSynchronize(stop);

// 计算耗时
float milliseconds = 0;
cudaEventElapsedTime(&milliseconds, start, stop);
printf("GPU Kernel time: %.4f ms\n", milliseconds);

// 清理
cudaEventDestroy(start);
cudaEventDestroy(stop);
```

**Event API 解释**（新人必看）：

| API | 作用 |
|-----|------|
| `cudaEventCreate(&e)` | 创建事件对象 |
| `cudaEventRecord(e)` | 在当前流中记录事件（异步，插入到流队列） |
| `cudaEventSynchronize(e)` | 阻塞 CPU 直到事件完成 |
| `cudaEventElapsedTime(&ms, start, stop)` | 计算两个事件之间的时间差（毫秒） |
| `cudaEventDestroy(e)` | 销毁事件对象 |

> 🔧 **为什么用 Event 而不是 CPU 的 chrono**：kernel 是异步的，CPU 的 `chrono` 测的是"CPU 下完命令到 CPU 收到完成通知"的时间，包含了 CPU 做其他事的时间，不准。Event 是 GPU 端打的时间戳，精确测量 GPU 实际执行时间。

### 性能结果对比（参考值）

在 RTX 4090 上，N = 2²⁴（约 1600 万元素）的典型结果：

| 📊 指标 | CPU (单核) | GPU (仅 Kernel) | GPU (含传输) |
|---------|-----------|-----------------|-------------|
| 耗时 | ~25 ms | ~0.3 ms | ~8 ms |
| 有效带宽 | ~5 GB/s | ~400 GB/s | ~15 GB/s |

⚠️ **注意**：向量加法是**内存密集型**操作（每元素只做 1 次加法但需要读 2 个 float、写 1 个 float）。含数据传输时 PCIe 带宽成为瓶颈。GPU 优势在数据已经在显存上时最明显。

### 性能分析

#### 算术强度（Arithmetic Intensity）

```
算术强度 = 计算量 / 内存访问量 = 1 FLOP / 12 Bytes ≈ 0.083 FLOP/Byte
```

这个值极低，说明向量加法是**典型的内存密集型**操作，性能受限于内存带宽而非计算能力。

#### PCIe 传输开销分析

| 📊 环节 | 典型耗时 | 占比 |
|---------|---------|------|
| H2D 传输 (64MB × 2) | ~5 ms | ~60% |
| Kernel 执行 | ~0.3 ms | ~4% |
| D2H 传输 (64MB) | ~3 ms | ~36% |

📌 **关键点**：对简单逐元素操作，数据传输是主要瓶颈。实际应用应尽量让数据"留在 GPU 上"——多个 Kernel 串联处理同一批数据，避免反复搬运。

#### 何时值得用 GPU

| 场景 | CPU 更好 | GPU 更好 |
|------|---------|---------|
| 数据量小（< 10K 元素） | ✅ | ❌ |
| 单次简单操作 + 数据不在 GPU | ✅ | ❌ |
| 大数据 + 数据已在 GPU | ❌ | ✅ |
| 大数据 + 多步串联计算 | ❌ | ✅ |
| 计算密集型（如矩阵乘法） | ❌ | ✅ |

### 进阶练习

#### 练习 1：向量点积（Reduce 预热）

实现 `result = Σ A[i] × B[i]`。需要**规约**——每线程计算部分乘积，Block 内用共享内存累加。（详见 `4. 算子实战篇/1. CUDA-Reduce算子优化.md`）

#### 练习 2：SAXPY

实现 `Y[i] = α × X[i] + Y[i]`，BLAS 基础操作，比向量加法多一个标量乘法。

#### 练习 3：Grid-stride Loop 版本

```cpp
__global__ void vectorAddStride(const float* A, const float* B,
                                float* C, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < N; i += stride) {
        C[i] = A[i] + B[i];
    }
}

// 启动时不再需要 gridSize = (N + 255) / 256
vectorAddStride<<<128, 256>>>(d_A, d_B, d_C, N);
```

#### 练习 4：异步传输与计算重叠

```cpp
cudaStream_t stream;
cudaStreamCreate(&stream);

cudaMemcpyAsync(d_A, h_A, size, cudaMemcpyHostToDevice, stream);
cudaMemcpyAsync(d_B, h_B, size, cudaMemcpyHostToDevice, stream);
vectorAdd<<<gridSize, blockSize, 0, stream>>>(d_A, d_B, d_C, N);
cudaMemcpyAsync(h_C, d_C, size, cudaMemcpyDeviceToHost, stream);

cudaStreamSynchronize(stream);
cudaStreamDestroy(stream);
```

💡 **提示**：要实现真正重叠，主机端内存必须用 `cudaMallocHost()`（Pinned Memory）分配。

## 面试回答（可直接口述的版本）

**问：请描述 CUDA 编程的基本流程，以及如何测量 Kernel 性能。**

答：CUDA 编程五步流程：先 `cudaMalloc` 分配显存，`cudaMemcpy` 把数据从 CPU 拷到 GPU，用 `<<<grid, block>>>` 启动 kernel，`cudaMemcpy` 把结果拷回 CPU，最后 `cudaFree` 释放。性能测量用 CUDA Event——`cudaEventRecord` 在 GPU 端打时间戳，`cudaEventElapsedTime` 算差值，精度微秒级。不能用 CPU 的 chrono，因为 kernel 是异步的。测完要算有效带宽验证是否打满——向量加法是内存密集型，算术强度只有 0.083 FLOP/Byte，瓶颈在带宽不在计算。含数据传输时 PCIe 是瓶颈，所以实际应用要让数据留在 GPU 上多步串联处理。

## 深入追问（可能被追问的点）

- **Q1: 为什么 `cudaMemcpy` 是同步的而 kernel 启动是异步的？**
  A: 设计权衡。`cudaMemcpy` 同步保证数据就绪后再继续；kernel 异步让 CPU 能继续准备下一任务。`cudaMemcpyAsync` 可以异步传输（配合 stream 和 pinned memory）。

- **Q2: Block 大小选 256 还是 512？**
  A: 向量加法寄存器用量少，256 和 512 都可以。用 `cudaOccupancyMaxPotentialBlockSize` 自动选最优值。512 可能因寄存器限制降低 occupancy。

- **Q3: 为什么有效带宽只有 670 GB/s，离 RTX 4090 的 1008 GB/s 峰值还差很多？**
  A: 向量加法太简单，kernel 启动开销和内存访问延迟占比大。计算密集型 kernel（如 GEMM）能接近峰值带宽。

- **Q4: Pinned Memory 为什么能加速传输？**
  A: 普通 malloc 的内存可能被换出到磁盘，GPU DMA 传输时需要先锁定页面，开销大。Pinned Memory 物理页面固定，DMA 直接访问，且支持异步传输与计算重叠。

- **Q5: 如何让数据传输和计算重叠？**
  A: 用 CUDA Stream + `cudaMemcpyAsync` + pinned memory。把数据分块，一块在传输时另一块在计算，形成流水线。

## 易混淆点对比

| 易混概念 | 区别 | 记忆技巧 |
|---------|------|----------|
| `cudaMalloc` vs `malloc` | 前者分配显存，后者分配主机内存 | cuda 前缀=GPU 端 |
| `cudaMemcpy` vs `cudaMemcpyAsync` | 前者同步阻塞，后者异步需配 stream | Async=异步需stream |
| `cudaGetLastError` vs `cudaDeviceSynchronize` | 前者查启动错误，后者查执行错误 | LastError=启动，Sync=执行 |
| CUDA Event vs CPU chrono | 前者 GPU 端精确，后者 CPU 端不准(异步) | Event=GPU，chrono=CPU |
| H2D vs D2H | 前者 CPU→GPU，后者 GPU→CPU | H=Host(CPU), D=Device(GPU) |
| kernel only vs end-to-end | 前者纯计算，后者含 PCIe 传输 | 简单操作传输是大头 |

## 常见报错速查表

| ❌ 报错 | 原因 | ✅ 解决 |
|--------|------|--------|
| `illegal memory access` | 越界访问或主机指针当设备指针 | 加 `if (idx < N)`，检查指针来源 |
| `out of memory` | 显存不足 | 减小 N 或分批处理 |
| kernel 没执行 | 启动后没同步就退出 | 加 `cudaDeviceSynchronize()` |
| 结果全是 0 | 改的是设备内存没拷回 | 加 `cudaMemcpy` D2H |
| `no kernel image available` | `-arch` 与 GPU 不匹配 | 用 `nvidia-smi --query-gpu=compute_cap` 查后重编译 |
| 验证失败 | 浮点误差或索引算错 | 放宽误差阈值，检查索引公式 |
| Event 计时为 0 | 没调用 `cudaEventSynchronize` | 在 `cudaEventElapsedTime` 前同步 |
