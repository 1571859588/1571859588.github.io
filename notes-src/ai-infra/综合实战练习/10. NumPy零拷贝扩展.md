# 练习 10：NumPy 零拷贝扩展——复制版 vs 共享缓冲区版

> 关联笔记：`综合实战：跨层思维与自检练习.md` 5.4 练习 10
> 知识点：pybind11::array_t、零拷贝、缓冲区协议、性能测量
> 难度：高

---

## 一句话结论

教学版用 Python list 输入（边界逐元素复制两次，可能比纯 Python 还慢）；生产版用 `pybind11::array_t<float>` 直接操作 NumPy 缓冲区指针——不复制数据，性能与 NumPy 相当。核心区别：list 版在边界上拆箱/装箱每个元素，array_t 版直接读写连续内存。

---

## 题目

改造综合实战的 `fast_scale`，让它接收 NumPy 数组，检查 dtype 与连续性，并比较"复制版"和"共享缓冲区版"的耗时。

---

## 知识点

| 概念 | 说明 |
|------|------|
| `pybind11::array_t<T>` | NumPy 数组的 C++ 视图，通过 `request()` 获取缓冲区信息 |
| `buf.request()` | 返回 `buffer_info`，包含 `ptr`（数据指针）、`shape`、`strides`、`itemsize` |
| 零拷贝 | 直接操作 `buf.ptr`，不复制数据 |
| 连续性检查 | `buf.strides[0] == sizeof(float)` 确保 C-contiguous |
| dtype 检查 | `array_t<float>` 模板参数保证 dtype，不需手动检查 |

---

### 第 2 步：C++ 实现（三个版本）

```cpp
// scale.cpp
#include <pybind11/pybind11.h>
#include <pybind11/numpy.h>
#include <pybind11/stl.h>
#include <vector>
#include <cstring>      // memcpy
#include <stdexcept>

namespace py = pybind11;

// ============================================================
// 版本 1：教学版（Python list → std::vector，逐元素复制）
// ============================================================
std::vector<float> scale_list(const std::vector<float>& input, float factor) {
    std::vector<float> output;
    output.reserve(input.size());
    for (float v : input) {
        output.push_back(v * factor);
    }
    return output;
    // 边界转换：list → vector（复制1次） + vector → list（复制2次）
}

// ============================================================
// 版本 2：复制版（NumPy array → 新数组，memcpy + 计算）
// ============================================================
py::array_t<float> scale_copy(py::array_t<float> input, float factor) {
    // 检查维度
    auto buf = input.request();
    if (buf.ndim != 1) {
        throw std::runtime_error("input must be 1-D array");
    }
    // 检查连续性（stride 应该等于元素大小）
    if (buf.strides[0] != sizeof(float)) {
        throw std::runtime_error("input must be contiguous (use np.ascontiguousarray)");
    }
    // dtype 由 array_t<float> 模板保证，不用检查

    size_t n = buf.shape[0];
    float* in_ptr = static_cast<float*>(buf.ptr);

    // 分配新输出数组
    auto output = py::array_t<float>(n);
    auto out_buf = output.request();
    float* out_ptr = static_cast<float*>(out_buf.ptr);

    // 复制 + 计算
    for (size_t i = 0; i < n; ++i) {
        out_ptr[i] = in_ptr[i] * factor;
    }

    return output;
    // 边界转换：无（直接操作指针），但分配了新数组
}

// ============================================================
// 版本 3：共享缓冲区版（直接修改输入数组，零拷贝）
// ============================================================
py::array_t<float> scale_inplace(py::array_t<float> input, float factor) {
    auto buf = input.request();
    if (buf.ndim != 1) {
        throw std::runtime_error("input must be 1-D array");
    }
    if (buf.strides[0] != sizeof(float)) {
        throw std::runtime_error("input must be contiguous");
    }

    float* ptr = static_cast<float*>(buf.ptr);
    size_t n = buf.shape[0];

    // 直接在原数组上修改（零拷贝，零分配）
    for (size_t i = 0; i < n; ++i) {
        ptr[i] *= factor;
    }

    return input;   // 返回原数组（被修改了）
    // 边界转换：无，分配：无——真正的零拷贝
}

// ============================================================
// 模块定义
// ============================================================
PYBIND11_MODULE(fast_scale, m) {
    m.def("scale_list",  &scale_list,  py::arg("input"), py::arg("factor"),
          "教学版：Python list 输入，逐元素复制");
    m.def("scale_copy",  &scale_copy,  py::arg("input"), py::arg("factor"),
          "复制版：NumPy 输入，分配新数组输出");
    m.def("scale_inplace", &scale_inplace, py::arg("input"), py::arg("factor"),
          "共享版：直接修改输入数组，零拷贝");
}
```

### 第 5 步：正确性测试

```python
# test_correctness.py
import sys
sys.path.insert(0, "build")

import numpy as np
import fast_scale

print("=== 正确性测试 ===")

# 测试数据
x_list = [1.0, -2.0, 3.5, 0.0, 100.0]
x_np = np.array(x_list, dtype=np.float32)
factor = 2.0
expected = [v * factor for v in x_list]

# 版本 1：list 版
result1 = fast_scale.scale_list(x_list, factor)
print(f"scale_list:    {result1}")
assert result1 == expected, f"list 版结果错误: {result1}"

# 版本 2：复制版
result2 = fast_scale.scale_copy(x_np, factor)
print(f"scale_copy:    {result2}")
assert np.allclose(result2, expected), f"复制版结果错误: {result2}"

# 版本 3：共享版（会修改输入！）
x_np_copy = x_np.copy()    # 先备份
result3 = fast_scale.scale_inplace(x_np_copy, factor)
print(f"scale_inplace: {result3}")
assert np.allclose(result3, expected), f"共享版结果错误: {result3}"
# 验证输入被修改了
assert np.allclose(x_np_copy, expected), "共享版应该修改了输入数组"

# 边界测试：空数组
empty = np.array([], dtype=np.float32)
assert fast_scale.scale_copy(empty, 2.0).size == 0
assert fast_scale.scale_inplace(empty, 2.0).size == 0
print("\n空数组测试通过")

# 边界测试：非连续数组
x_noncontig = np.array([[1, 2, 3], [4, 5, 6]], dtype=np.float32)[0, :]
# x_noncontig 是连续的（一维切片），但如果是列切片就不连续
x_col = np.array([[1, 2, 3], [4, 5, 6]], dtype=np.float32)[:, 0]
print(f"\n列切片是否连续: {x_col.flags['C_CONTIGUOUS']}")  # False
try:
    fast_scale.scale_copy(x_col, 2.0)
    print("错误：应该抛异常")
except RuntimeError as e:
    print(f"正确捕获非连续数组: {e}")

# 修复：用 ascontiguousarray
x_contig = np.ascontiguousarray(x_col)
result = fast_scale.scale_copy(x_contig, 2.0)
print(f"修复后: {result}")

print("\n=== 所有正确性测试通过 ===")
```

### 第 6 步：性能对比

```python
# benchmark.py
import sys
import time
sys.path.insert(0, "build")

import numpy as np
import fast_scale

def benchmark(sizes):
    print(f"{'size':>12} | {'list版':>10} | {'copy版':>10} | {'inplace版':>10} | {'numpy':>10}")
    print("-" * 65)

    for n in sizes:
        # 准备数据
        x_list = [float(i) for i in range(n)]
        x_np = np.array(x_list, dtype=np.float32)
        factor = 2.0

        # 预热（让缓存热起来）
        if n == sizes[0]:
            fast_scale.scale_list(x_list[:100], factor)
            fast_scale.scale_copy(x_np[:100], factor)
            fast_scale.scale_inplace(x_np[:100].copy(), factor)
            x_np[:100] * factor

        # 版本 1：list 版
        t0 = time.perf_counter()
        for _ in range(10):  # 多跑几次取平均
            fast_scale.scale_list(x_list, factor)
        t_list = (time.perf_counter() - t0) / 10

        # 版本 2：copy 版
        t0 = time.perf_counter()
        for _ in range(100):
            fast_scale.scale_copy(x_np, factor)
        t_copy = (time.perf_counter() - t0) / 100

        # 版本 3：inplace 版
        t0 = time.perf_counter()
        for _ in range(100):
            x_copy = x_np.copy()       # 每次复制一份（公平对比）
            fast_scale.scale_inplace(x_copy, factor)
        t_inplace = (time.perf_counter() - t0) / 100

        # NumPy 基准
        t0 = time.perf_counter()
        for _ in range(100):
            x_np * factor
        t_numpy = (time.perf_counter() - t0) / 100

        print(f"{n:>12,} | {t_list*1000:>8.2f}ms | {t_copy*1000:>8.2f}ms | {t_inplace*1000:>8.2f}ms | {t_numpy*1000:>8.2f}ms")

benchmark([1_000, 10_000, 100_000, 1_000_000, 10_000_000])
```

#### 预期结果

```
        size |     list版 |    copy版 | inplace版 |     numpy
-----------------------------------------------------------------
       1,000 |     0.12ms |     0.01ms |     0.01ms |     0.00ms
      10,000 |     1.15ms |     0.03ms |     0.02ms |     0.01ms
     100,000 |    11.50ms |     0.25ms |     0.18ms |     0.08ms
   1,000,000 |   115.00ms |     2.50ms |     1.80ms |     0.80ms
  10,000,000 |  1150.00ms |    25.00ms |    18.00ms |     8.00ms
```

> 实际数值因 CPU 和内存带宽而异，但**趋势不变**：
> - **list 版最慢**（比 NumPy 慢 100x+）：边界逐元素拆箱/装箱开销巨大
> - **copy 版比 list 版快 50x**：没有边界转换，但多了一次数组分配 + memcpy
> - **inplace 版比 copy 版快 30%**：零分配，直接修改输入缓冲区
> - **NumPy 最快**：内部用 SIMD + 高度优化的 BLAS

## 逐行解释

### 三个版本的边界转换对比

```
版本 1（list 版）:
  Python list → [逐元素拆箱 float→C++ float] → std::vector → 计算 → std::vector → [逐元素装箱 C++→Python] → Python list
                  ↑ O(n) 复制 + 对象开销                                    ↑ O(n) 复制 + 对象开销

版本 2（copy 版）:
  NumPy array → buf.request() → 拿到 float* 指针 → 计算 → 新 array_t → 返回
                ↑ 零转换（直接拿指针）            ↑ 一次分配

版本 3（inplace 版）:
  NumPy array → buf.request() → 拿到 float* 指针 → 直接修改 → 返回原数组
                ↑ 零转换                      ↑ 零分配
```

### `buf.request()` 返回什么

```cpp
auto buf = input.request();
// buf 是 py::buffer_info，包含：
//   buf.ptr       → void*，数据起始指针
//   buf.shape     → vector<ssize_t>，每个维度的大小
//   buf.strides   → vector<ssize_t>，每个维度的步长（字节）
//   buf.ndim      → int，维度数
//   buf.itemsize  → ssize_t，每个元素的字节数
```

### 为什么检查 `buf.strides[0] == sizeof(float)`？

NumPy 数组不一定在内存中连续。例如：

```python
a = np.array([[1, 2, 3], [4, 5, 6]], dtype=np.float32)  # 2x3，C-contiguous
b = a[:, 0]    # 取第一列：[1, 4]
```

`b` 的数据在内存中不连续——`b[0]` 在地址 0，`b[1]` 在地址 12（跳了 3 个 float = 12 字节），而不是 4 字节。

`buf.strides[0]` = 12（不等于 `sizeof(float)` = 4），说明不连续。直接用 `ptr[i]` 访问会读错位置。

**修复**：调用方用 `np.ascontiguousarray()` 转成连续数组再传入。

### 为什么 inplace 版要 `x_np.copy()` 后再传？

```python
x_copy = x_np.copy()       # 公平对比：也算上这次复制的开销
fast_scale.scale_inplace(x_copy, factor)
```

因为 inplace 版会**修改输入数组**。如果不 copy，第二次循环拿到的已经是乘过 factor 的数组，再乘一次就变成 factor² 了。copy 一次保证每次循环都从原始数据开始。

如果使用场景允许修改输入，inplace 版不需要 copy，会更快。

## 三个版本怎么选

| 版本 | 适用场景 | 优点 | 缺点 |
|------|----------|------|------|
| `scale_list` | 教学/原型 | 简单直观 | 慢（边界转换开销大） |
| `scale_copy` | 生产（不改输入） | 快，输入安全 | 多一次分配 |
| `scale_inplace` | 生产（允许改输入） | 最快，零分配 | 修改了输入，调用方要注意 |

---

## 常见错误

| 错误 | 原因 | 修复 |
|------|------|------|
| `TypeError: incompatible function arguments` | 传了 Python list 给 `array_t` 版本 | 传 `np.array(..., dtype=np.float32)` |
| 结果全是垃圾值 | 输入不是 float32（如 float64） | 确认 `dtype=np.float32`，或用模板支持多类型 |
| 非连续数组报错 | NumPy 切片产生非连续视图 | `np.ascontiguousarray(x)` 后再传 |
| inplace 版结果不对 | 输入被修改了但调用方不知道 | 文档明确标注"会修改输入"，或只返回新数组 |
| list 版比纯 Python 还慢 | 边界转换 O(n) + 装箱开销 > 循环节省 | 换 array_t 版本 |
